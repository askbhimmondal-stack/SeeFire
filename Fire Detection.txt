const FIRE_THRESHOLD = 0.08; // 8 % of pixels must be "fire-colored"

/* ── Alarm Control (MP3) ────────────────────────────────── */
/**
 * Plays the external MP3 alarm sound in a loop.
 * Uses the HTML <audio> element with loop attribute.
 */
function startAlarm() {
  if (alarmPlaying) return;
  alarmPlaying = true;
  alarmAudio.volume = 1.0;
  alarmAudio.currentTime = 0;
  alarmAudio.play().catch(err => console.warn('Audio play blocked:', err));
}

/**
 * Stops and resets the alarm sound.
 */
function stopAlarm() {
  if (!alarmPlaying) return;
  alarmPlaying = false;
  alarmAudio.pause();
  alarmAudio.currentTime = 0;
}

/* ── Color Analysis ─────────────────────────────────────── */
/**
 * Determines whether a pixel (r,g,b) looks like fire/flame.
 * Fire colors are characterised by:
 *   - High red channel
 *   - Medium-to-high green channel (orange/yellow component)
 *   - Low blue channel
 */
function isFirePixel(r, g, b) {
  return (
    r > 190 &&       // strong red
    g > 60  &&       // some green (orange / yellow range)
    g < 200 &&       // but not too green (avoids white)
    b < 80  &&       // very little blue
    r > g + 40       // red dominates green
  );
}

/**
 * Analyse a single video frame on the hidden canvas.
 * Returns the fraction of pixels that are fire-colored (0..1).
 */
function analyzeFrame() {
  const ctx   = analyzeCanvas.getContext('2d');
  const w     = analyzeCanvas.width;
  const h     = analyzeCanvas.height;
  ctx.drawImage(video, 0, 0, w, h);

  const imageData = ctx.getImageData(0, 0, w, h);
  const data      = imageData.data;
  let firePixels  = 0;
  const total     = w * h;

  for (let i = 0; i < data.length; i += 4) {
    if (isFirePixel(data[i], data[i+1], data[i+2])) {
      firePixels++;
      // Optionally tint fire pixels for overlay — store position
    }
  }
  return firePixels / total;
}

/**
 * Draw a heat-highlight overlay on the visible canvas.
 * Paints semi-transparent red over pixels that look like fire.
 */
function drawOverlay() {
  const ctx  = overlay.getContext('2d');
  const w    = overlay.width;
  const h    = overlay.height;
  const ac   = analyzeCanvas.getContext('2d');
  const data = ac.getImageData(0, 0, analyzeCanvas.width, analyzeCanvas.height).data;
  const scaleX = w / analyzeCanvas.width;
  const scaleY = h / analyzeCanvas.height;

  ctx.clearRect(0, 0, w, h);

  // Draw fire highlights
  const aw = analyzeCanvas.width;
  for (let i = 0; i < data.length; i += 4) {
    if (isFirePixel(data[i], data[i+1], data[i+2])) {
      const px = (i/4) % aw;
      const py = Math.floor((i/4) / aw);
      ctx.fillStyle = 'rgba(255,80,0,0.35)';
      ctx.fillRect(px * scaleX, py * scaleY, scaleX + 1, scaleY + 1);
    }
  }
}

/* ── Main Detection Loop ────────────────────────────────── */

function detectionLoop(timestamp) {
  // FPS counter
  if (lastFrameTime) {
    const fps = Math.round(1000 / (timestamp - lastFrameTime));
    fpsLabel.textContent = fps + ' FPS';
  }
  lastFrameTime = timestamp;

  // Only process if video is playing
  if (video.readyState >= 2) {
    const ratio = analyzeFrame();
    const pct   = Math.min(ratio * 100, 100).toFixed(1);

    // Update heat bar
    heatFill.style.width = Math.min(ratio / FIRE_THRESHOLD * 100, 100) + '%';
    heatPct.textContent  = pct + '%';
    statFire.textContent = pct + '%';
    statFire.className   = ratio >= FIRE_THRESHOLD ? 'value hot' : 'value';

    const nowFire = ratio >= FIRE_THRESHOLD;

    if (nowFire && !fireDetected) {
      // Transition to fire state
      fireDetected = true;
      alertCount++;
      statAlerts.textContent = alertCount;
      setFireUI(true);
      stopAlarmBtn.disabled = false;
    } else if (!nowFire && fireDetected) {
      // Transition to safe state
      fireDetected = false;
      setFireUI(false);
      stopAlarm();
      stopAlarmBtn.disabled = true;
    }

    // Start alarm if fire detected and not already playing
    if (fireDetected) {
      if (!alarmPlaying) startAlarm();
      drawOverlay();
    } else {
      overlay.getContext('2d').clearRect(0, 0, overlay.width, overlay.height);
    }
  }

  animFrame = requestAnimationFrame(detectionLoop);
}

/* ── Camera Control ─────────────────────────────────────── */
startBtn.addEventListener('click', async () => {
  // If already running, stop camera
  if (stream) {
    stopCamera();
    return;
  }

  try {
    stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: 'environment' }, audio: false });
    video.srcObject = stream;
    placeholder.style.display = 'none';

    video.addEventListener('loadedmetadata', () => {
      // Match analysis canvas to a small resolution for performance
      analyzeCanvas.width  = 160;
      analyzeCanvas.height = 90;
      // Match overlay canvas to display size
      overlay.width  = video.videoWidth  || 640;
      overlay.height = video.videoHeight || 360;
    });

    recDot.classList.add('live');
    recLabel.textContent = 'LIVE';
    statusBanner.className  = 'status-banner safe';
    statusBanner.textContent= '✅ No Fire Detected';
    statStatus.textContent  = 'CLEAR';
    statStatus.className    = 'value safe';
    startBtn.textContent    = '⏹ Stop Camera';

    animFrame = requestAnimationFrame(detectionLoop);
  } catch (err) {
    statusBanner.className  = 'status-banner idle';
    statusBanner.textContent= '❌ Camera Access Denied: ' + err.message;
    console.error('Camera error:', err);
  }
});

function stopCamera() {
  if (stream) {
    stream.getTracks().forEach(t => t.stop());
    stream = null;
  }
  video.srcObject    = null;
  placeholder.style.display = 'flex';
  if (animFrame) cancelAnimationFrame(animFrame);
  fireDetected = false;
  stopAlarm();
  recDot.classList.remove('live');
  recLabel.textContent    = 'STANDBY';
  fpsLabel.textContent    = '-- FPS';
  heatFill.style.width    = '0%';
  heatPct.textContent     = '0%';
  statFire.textContent    = '0%';
  startBtn.textContent    = '▶ Start Camera';
  stopAlarmBtn.disabled   = true;
  setOfflineUI();
  overlay.getContext('2d').clearRect(0,0,overlay.width,overlay.height);
}
